```python
# test/test_dependencies.py
import osqp
import scipy.sparse as sp
import numpy as np
from skimage.morphology import skeletonize

def test_osqp():
    """测试OSQP安装"""
    P = sp.eye(2, format='csc')
    q = np.ones(2)
    A = sp.eye(2, format='csc')
    l = np.zeros(2)
    u = np.ones(2)
    
    prob = osqp.OSQP()
    prob.setup(P, q, A, l, u, verbose=False)
    result = prob.solve()
    
    assert result.info.status == 'solved', "OSQP求解失败"
    print("✅ OSQP测试通过")

def test_distance_transform():
    """测试距离场计算"""
    from scipy.ndimage import distance_transform_edt
    
    grid = np.zeros((100, 100))
    grid[40:60, 40:60] = 1  # 障碍物
    
    dist = distance_transform_edt(1 - grid)
    
    assert dist.shape == (100, 100), "距离场形状错误"
    assert dist[50, 50] == 0, "障碍物内部距离应为0"
    assert dist[0, 0] > 0, "自由空间距离应>0"
    print("✅ 距离场计算测试通过")

def test_skeletonization():
    """测试骨架提取"""
    grid = np.zeros((100, 100))
    grid[10:90, 45:55] = 1  # 长条形通道
    
    skeleton = skeletonize(grid)
    
    assert skeleton.sum() > 0, "骨架应非空"
    print("✅ 骨架提取测试通过")

if __name__ == '__main__':
    test_osqp()
    test_distance_transform()
    test_skeletonization()
    print("\n🎉 所有依赖包测试通过！")
```

### 4.1 修改env.py

#### 4.1.1 初始化部分

在`NavigationEnv.__init__`中添加：

```python
# training/scripts/env.py
class NavigationEnv:
    def __init__(self, cfg):
        super().__init__()
        
        # ... 原有初始化 ...
        
        # ========== 新增：拓扑模块初始化 ==========
        if cfg.topo.use_topo:
            from modules.topo_extractor import TopoExtractor
            from modules.graph_transformer import GraphTransformer
            from modules.safety_shield import SafetyShieldQP
            from modules.hierarchical_control import HierarchicalController
            
            # 拓扑提取器
            self.topo_extractor = TopoExtractor(cfg.topo, device=self.device)
            
            # 图策略（作为actor）
            self.graph_policy = GraphTransformer(cfg.topo).to(self.device)
            
            # 安全盾
            if cfg.topo.use_safety_shield:
                self.safety_shield = SafetyShieldQP(
                    relaxation_weight=cfg.topo.qp_relaxation_weight,
                    max_relaxation=cfg.topo.qp_max_relaxation,
                    v_max=cfg.topo.qp_v_max,
                    safe_distance=cfg.topo.safe_distance,
                )
            
            # 分层控制器
            if cfg.topo.use_hierarchical:
                self.hierarchical_controller = HierarchicalController(
                    high_freq=cfg.sim.freq,
                    low_freq=cfg.topo.planning_freq,
                    pid_gains={
                        'kp': cfg.topo.pid_kp,
                        'kd': cfg.topo.pid_kd,
                        'ki': cfg.topo.pid_ki,
                    },
                    goal_horizon=cfg.topo.goal_horizon,
                )
    
    def reset(self):
        """重置环境"""
        obs = super().reset()
        
        # 重置分层控制器
        if hasattr(self, 'hierarchical_controller'):
            self.hierarchical_controller.reset(self.num_envs, self.device)
        
        # 重置统计
        if hasattr(self, 'safety_shield'):
            self.safety_shield.reset_statistics()
        
        return obs
```

#### 4.1.4 _compute_state_and_obs修改

```python
def _compute_state_and_obs(self, obs_buf):
    """计算状态和观测"""
    # ... 原有LiDAR、状态计算 ...
    
    # ========== 新增：拓扑提取 ==========
    if hasattr(self, 'topo_extractor'):
        graph_data = self.topo_extractor.extract_batch(
            ray_hits_w=self.lidar.data.ray_hits_w.reshape(self.num_envs, -1, 3),
            ray_pos_w=self.lidar.data.pos_w[:, 0, :],
            ego_vel=self.drone.vel,
            goal_pos=self.target_pos,
        )
        
        # 添加到观测
        obs_buf["graph_node_positions"] = graph_data['node_positions']
        obs_buf["graph_node_features"] = graph_data['node_features']
        obs_buf["graph_edge_features"] = graph_data['edge_features']
        obs_buf["graph_node_mask"] = graph_data['node_mask']
        obs_buf["graph_edge_mask"] = graph_data['edge_mask']
    
    return obs_buf
```

#### 4.1.5 _pre_sim_step修改

**这是核心修改**，整合所有新模块：

```python
def _pre_sim_step(self, tensordict):
    """在仿真step前处理动作"""
    
    # ========== 1. 判断是否高层更新 ==========
    is_high_update = hasattr(self, 'hierarchical_controller') and \
                     (self.step_counter % self.hierarchical_controller.freq_ratio == 0)
    
    # ========== 2. 高层规划（仅在更新时刻） ==========
    # 关键：这里必须消费 PPO 传入的离散动作，不能在 env 内再次采样，
    # 否则执行动作与 log_prob 对不齐，PPO 梯度目标失效。
    if is_high_update:
        node_indices = tensordict["action"].squeeze(-1).long()  # (batch,)

        # 将节点动作映射到速度
        high_level_velocity = self._node_to_velocity(
            node_indices,
            tensordict["graph_node_positions"],  # (batch, N+1, 3), index 0=ego
            self.drone.pos,
        )
    else:
        high_level_velocity = None
    
    # ========== 3. 低层控制（每步） ==========
    if hasattr(self, 'hierarchical_controller'):
        control_velocity, ctrl_info = self.hierarchical_controller.step(
            ego_pos=self.drone.pos,
            ego_vel=self.drone.vel,
            high_level_velocity=high_level_velocity,
        )
        
        # 记录跟踪误差（用于奖励）
        tensordict["tracking_error"] = ctrl_info['tracking_error']
    else:
        # 无分层控制，直接使用high_level_velocity
        if high_level_velocity is not None:
            control_velocity = high_level_velocity
        else:
            # Fallback：使用原CNN策略输出
            control_velocity = tensordict["action"]
    
    # ========== 4. 安全盾 ==========
    if hasattr(self, 'safety_shield'):
        # 提取障碍物信息
        dyn_obs = self._get_dyn_obs_tensor()
        
        # QP优化
        safe_velocity, intervention = self.safety_shield.solve(
            control_velocity,
            dyn_obs,
            self.drone.pos,
        )
        
        # 记录干预（用于奖励）
        tensordict["intervention"] = intervention
        tensordict["action"] = safe_velocity
    else:
        tensordict["action"] = control_velocity
    
    # ========== 5. 应用动作 ==========
    self.drone.apply_action(tensordict["action"])
    
    return tensordict

def _node_to_velocity(self, node_indices, node_positions, ego_pos, v_target=2.0, dt_low=0.1):
    """将选中的节点索引转换为速度向量"""
    batch_size = node_indices.shape[0]
    device = ego_pos.device

    # Categorical 索引是候选节点 [0, N-1]，真实节点索引需要 +1（0 保留给 ego）
    real_idx = (node_indices + 1).clamp(1, node_positions.shape[1] - 1)
    selected_pos = node_positions[torch.arange(batch_size, device=device), real_idx]  # (batch, 3)

    direction = selected_pos - ego_pos
    dist = torch.norm(direction, dim=-1, keepdim=True).clamp(min=1e-4)
    direction_unit = direction / dist

    # 速度幅值限制：不超过巡航速度，且一步内不越过目标节点
    v_mag = torch.minimum(
        torch.full_like(dist, v_target),
        dist / dt_low,
    )
    velocities = direction_unit * v_mag

    return velocities

def _get_dyn_obs_tensor(self):
    """提取动态障碍物的7维向量表示"""
    # （根据实际环境实现）
    # 示例：假设有self.obstacles属性
    
    num_envs = self.num_envs
    max_obs = 10
    
    obs_tensor = torch.zeros(num_envs, max_obs, 7, device=self.device)
    
    if hasattr(self, 'obstacles') and self.obstacles is not None:
        for env_idx in range(num_envs):
            obs_list = self.obstacles[env_idx]  # List of obstacle objects
            
            for i, obs in enumerate(obs_list[:max_obs]):
                obs_tensor[env_idx, i, :3] = obs.pos
                obs_tensor[env_idx, i, 3:6] = obs.vel
                obs_tensor[env_idx, i, 6] = obs.radius
    
    return obs_tensor
```

```python
# training/scripts/ppo.py
class PPO:
    def collect_rollouts(self, env, num_steps):
        """收集rollout数据"""
        rollout_data = []
        
        obs = env.reset()
        
        for step in range(num_steps):
            # ========== 原逻辑 ==========
            # action, log_prob, value = self.policy(obs)
            
            # ========== 新逻辑：支持图数据 ==========
            if self.cfg.mode == 'graph_ppo':
                # 提取图数据
                graph_data = {
                    'node_features': obs["graph_node_features"],
                    'edge_features': obs["graph_edge_features"],
                    'node_mask': obs["graph_node_mask"],
                    'edge_mask': obs["graph_edge_mask"],
                }
                
                # 使用图策略
                node_probs, _ = self.graph_policy(
                    graph_data['node_features'],
                    graph_data['edge_features'],
                    graph_data['node_mask'],
                    graph_data['edge_mask'],
                )
                
                # 采样动作（节点索引）
                action_dist = torch.distributions.Categorical(node_probs)
                action = action_dist.sample()
                log_prob = action_dist.log_prob(action)
                
                # Value（使用ego节点特征）
                ego_features = graph_data['node_features'][:, 0, :]  # 假设节点0是ego
                value = self.critic(ego_features)
            else:
                # 原CNN策略
                action, log_prob, value = self.policy(obs)
            
            # 环境步进
            next_obs, reward, done, info = env.step(action)
            
            # 存储
            rollout_data.append({
                'obs': obs,
                'action': action,
                'log_prob': log_prob,
                'reward': reward,
                'value': value,
            })
            
            obs = next_obs
        
        return rollout_data
```

#### 5.1.2 Critic网络修改

原Critic可能使用CNN处理LiDAR，现在使用图特征：

```python
class GraphCritic(nn.Module):
    """基于图数据的Critic"""
    
    def __init__(self, node_feat_dim=18, hidden_dim=128):
        super().__init__()
        
        self.mlp = nn.Sequential(
            nn.Linear(node_feat_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, 1),
        )
    
    def forward(self, ego_node_features):
        """
        计算状态值
        
        Args:
            ego_node_features: (batch, node_feat_dim) ego节点特征
        
        Returns:
            value: (batch, 1) 状态值
        """
        return self.mlp(ego_node_features)
```

1. 提高PID增益：`pid_kp=3.0, pid_kd=1.0`
2. 缩短目标视野：`goal_horizon=0.05`
3. 提高规划频率：`planning_freq=15.0`
4. 检查是否高层输出速度过大

### Q5: 训练不收敛

**症状**：reward长时间不上升

**诊断**：
1. 检查奖励设计：各项奖励权重是否合理
2. 检查策略输出：`node_probs`是否合理（不应过度集中）
3. 检查Critic：Value估计是否准确

**解决**：
1. 调整奖励权重：
   ```yaml
   reward_goal_weight: 1.0
   reward_collision_weight: -5.0  # 从-10降到-5（避免过度惩罚）
   reward_intervention_weight: -0.1  # 从-0.2降到-0.1
   ```
2. 增加探索：
   ```python
   # 在采样时添加噪声
   node_probs = node_probs + torch.rand_like(node_probs) * 0.1
   node_probs = node_probs / node_probs.sum(dim=-1, keepdim=True)
   ```
3. 降低学习率：`ppo.lr=1e-4`
